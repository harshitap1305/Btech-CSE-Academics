Conclusion: 
From the experiments, it was observed that indexing does not always guarantee faster query execution. In our case, since all students had the same major ('Engineering'), the index on the major column did not improve performance and even made the query slightly slower due to low data selectivity. However, insertion operations were noticeably slower when indexes were present, as maintaining index structures increases overhead. Overall, indexes are beneficial for queries on columns with diverse values and frequent searches but should be used carefully, as they can degrade insertion and update performance.

import sqlite3
import time

DB_FILE = "university.db"
QUERY = "SELECT * FROM Students WHERE major = 'Engineering';"
ITERATIONS = 100

con = sqlite3.connect(DB_FILE)
cur = con.cursor()

# Drop index if it exists
try:
    cur.execute("DROP INDEX idx_students_major")
    print("Index dropped.")
except sqlite3.OperationalError:
    print("Index did not exist.")

total_time = 0
for _ in range(ITERATIONS):
    start_time = time.time()
    cur.execute(QUERY).fetchall()
    end_time = time.time()
    total_time += (end_time - start_time)

print(f"Avg. time without index: {total_time / ITERATIONS:.6f} seconds")
con.close()

import sqlite3
import random
import time

DB_FILE = "university.db"
con = sqlite3.connect(DB_FILE)
cur = con.cursor()

# Ensure indexes exist
cur.execute("CREATE INDEX IF NOT EXISTS idx_enrollments_student_id ON Enrollments(student_id)")
cur.execute("CREATE INDEX IF NOT EXISTS idx_enrollments_course_id ON Enrollments(course_id)")
print("Indexes ensured.")

start_time = time.time()

for _ in range(500):
    student_id = random.randint(1, 2000)
    course_id = random.randint(1, 100)
    grade = round(random.uniform(2.0, 4.0), 2)
    cur.execute("INSERT INTO Enrollments (student_id, course_id, grade) VALUES (?, ?, ?)",
                (student_id, course_id, grade))

con.commit()
end_time = time.time()

print(f"Time to insert 500 records (with indexes): {end_time - start_time:.6f} seconds")
